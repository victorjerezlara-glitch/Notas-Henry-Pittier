// secciones-clonar/index.ts
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";
import { verifyJWT } from "https://esm.sh/jose@4.14.4";

Deno.serve(async (req) => {
  try {
    if (req.method !== "POST") {
      return new Response(
        JSON.stringify({ error: "Método no permitido" }),
        { status: 405 }
      );
    }

    // 1. Validar token
    const token = req.headers.get("Authorization")?.replace("Bearer ", "");
    if (!token) {
      return new Response(JSON.stringify({ error: "Token requerido" }), {
        status: 401,
      });
    }

    const { payload } = await verifyJWT(token, Deno.env.get("JWT_SECRET"), {
      algorithms: ["HS256"],
    });

    const user_id = payload.sub;

    // 2. Validar roles permitidos
    const supabaseAdmin = createClient(
      Deno.env.get("SUPABASE_URL"),
      Deno.env.get("SUPABASE_SECRET_KEYS")
    );

    const { data: roles } = await supabaseAdmin
      .from("user_roles")
      .select("id_rol")
      .eq("user_id", user_id);

    const rolesPermitidos = [1, 2, 4]; // superadmin, directivo, control_estudios
    const tienePermiso = roles?.some((r) =>
      rolesPermitidos.includes(r.id_rol)
    );

    if (!tienePermiso) {
      return new Response(
        JSON.stringify({
          error: "No tiene permisos para clonar secciones"
        }),
        { status: 403 }
      );
    }

    // 3. Obtener datos del body
    const body = await req.json();
    const { id_seccion_origen, nombre_nuevo, letra_nueva } = body;

    if (!id_seccion_origen || !nombre_nuevo || !letra_nueva) {
      return new Response(
        JSON.stringify({
          error: "id_seccion_origen, nombre_nuevo y letra_nueva son obligatorios"
        }),
        { status: 400 }
      );
    }

    // 4. Llamar a la función SQL
    const { data, error } = await supabaseAdmin.rpc("clonar_seccion", {
      p_id_seccion_origen: id_seccion_origen,
      p_nombre_nuevo: nombre_nuevo,
      p_letra_nueva: letra_nueva
    });

    if (error) {
      return new Response(JSON.stringify({ error: error.message }), {
        status: 400,
      });
    }

    return new Response(JSON.stringify(data), { status: 200 });

  } catch (err) {
    return new Response(JSON.stringify({ error: err.message }), {
      status: 500,
    });
  }
});
